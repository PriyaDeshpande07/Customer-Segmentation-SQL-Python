SELECT AVG(SAL) FROM EMP;

SELECT DEPTNO, AVG(SAL)
FROM EMP
GROUP BY DEPTNO;

SELECT DEPTNO, COUNT(*)
FROM EMP
GROUP BY DEPTNO;

SELECT ENAME, SAL
FROM EMP
ORDER BY SAL DESC;

SELECT E.ENAME, D.DNAME
FROM EMP E
JOIN DEPT D
ON E.DEPTNO = D.DEPTNO;

SELECT ENAME,
       SAL,
       RANK() OVER (ORDER BY SAL DESC) AS SALARY_RANK
FROM EMP;
